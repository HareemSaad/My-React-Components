initialHeight='400px' // height of blob
initialWidth='400px' // width of blob
shrink='false' // disables shrinking/growing of blob
newHeight='200px' // shrinks the blob to this height | if > initial height, blob grows
newWidth='200px' // shrinks the blob to this width
//x y set for img 600 x 600 px
x='-10%' //starting point of image
y='-10%' //staring point of image
imgBlob='true' // if blob will render image or color
